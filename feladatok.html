1.

Hozz létre egy listát szamok néven, melyben 20db, -25 … 100 közötti egész szám van.
import random

szamok = [] #létrehoztam az üres listát
i = 0 # csináltam egy változót
while i < 20:
  szam = random.randint(-25, 100) #random számokat generáltam
  szamok.append(szam) #beírtam a számokat a listába
  i += 1 #lezártam a ciklust

2.

Írasd ki a lista elemeit egy sorba pontosvesszővel elválasztva
szamok_str = [] #egy lista amibe str-ként kezeli a számokat
i = 0
while i < 20:
 szamok_str.append(str(szamok[i])) #feltölti az eredeti számokat, msotmár str-ként az új listába
 i += 1
print(';'.join(szamok_str))

3.

Kérj be 2 számot, írd ki melyik a nagyobb és mennyivel
szam1 = int(input("Kérek egy számot:")) #bekértem a számokat
szam2 = int(input("Kérek egy másik számot:"))
if szam1 < szam2: # ha a szam1 kissebb mint a szam2 akkor ez a parancs teljesül
 print("A kissebb szám:", szam1)
 print("A két szám különbsége:", szam2 - szam1)
else: #hogyha nagyobb akkor pedig ez
 print("A kissebbb szám:", szam2)
 print("A két szám különbsége:", szam1-szam2)
